(interval 1 sec)

            for (var i = 1; i <= 5; i++) {
                setTimeout(function() {
                    console.log(i);
                }, 1000 * i);
            }

Rešenje uz ES2016

            for (let i = 1; i <= 5; i++) {
                setTimeout(function() {
                    console.log(i);
                }, 1000 * i);
            }

Rešenje

            for (var i = 1; i <= 5; i++) {
                setTimeout(function(x) {
                    return function() {
                        console.log(x);
                    };
                }(i), 1000 * i);
            }

Još jedno rešenje

            for (var i = 1; i <= 5; i++) {
                (function(x) {
                    setTimeout(function() {
                        console.log(x);
                    }, 1000 * x);
                })(i);
            }

Rešenje uz bind

            for (var i = 1; i <= 5; i++) {
                setTimeout((function(x) {
                    console.log(x);
                }).bind(null, i), i * 1000);
            }

Popraviti kod

            function Person (name) {
                if (name) {
                    this.options.name = name;
                }
            };
            Person.prototype.options = {
                name: 'Default name'
            };
            var foo = new Person('foo');
            var bar = new Person('bar');
            console.log(foo.options.name); // bar
            console.log(bar.options.name); // bar

Rešenje

            function Person (name) {
                if (name) {
                    this.options = { name };
                }
            };
            Person.prototype.options = {
                name: 'Default name'
            };
            var foo = new Person('foo');
            var bar = new Person('bar');
            console.log(foo.options.name); // foo
            console.log(bar.options.name); // bar
